def self.get_revisions(path)
  logcmd = %Q[git log -n1 --grep="^ *git-svn-id: .*@[0-9][0-9]* "]
  idpat = /git-svn-id: .*?@(\d+) \S+\Z/
  last = %x#{logcmd}`[idpat, 1]
  if path
    log = %x#{logcmd} "#{path}"`
    changed = log[idpat, 1]
    modified = %xgit log --format=%ai -- #{path}`
  else
    changed = last
  end
  [last, changed, modified]
end
